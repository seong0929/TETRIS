package org.teamseven.tetris.block.item;

import org.teamseven.tetris.Board.GameBoard;
import org.teamseven.tetris.block.CurrBlock;
import org.teamseven.tetris.handler.GameHandler;

import java.awt.*;
import java.util.Random;

public class ColorRemoveBlock extends ItemUnitBlock{

    private Color deleteColor;

    public ColorRemoveBlock() {
        super(Color.WHITE);
        Random random = new Random();

        int index = random.nextInt(7);
        switch (index) {
            case 0:
                deleteColor = Color.CYAN;
                break;
            case 1:
                deleteColor = Color.BLUE;
                break;
            case 2:
                deleteColor = Color.ORANGE;
                break;
            case 3:
                deleteColor = Color.YELLOW;
                break;
            case 4:
                deleteColor = Color.GREEN;
                break;
            case 5:
                deleteColor = Color.MAGENTA;
                break;
            case 6:
                deleteColor = Color.RED;
        }
    }

    @Override
    public void execute(GameBoard board, CurrBlock curr, GameHandler handler) {
        int cnt = 0;

        for (int j = 0; j < curr.height(); j++) {
            for (int i = 0; i < curr.width(); i++) {
                if (board.getBoard()[j][i].getColor() == deleteColor) {
                    cnt++;
                }
            }
        }

    }

    @Override
    public int execute(GameBoard board, CurrBlock curr, int[] pos) {
        return 0;
    }
}
